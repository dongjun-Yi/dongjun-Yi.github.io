def removeDuplicateLetters(self, s: str) -> str:
        counter, seen, stack = collections.Counter(s), set(), []

        for char in s:
            counter[char] -= 1
            if char in seen:
                continue
            # ë’¤ì— ë¶™ì¼ ë¬¸ìê°€ ë‚¨ì•„ ìˆë‹¤ë©´ ìŠ¤íƒì—ì„œ ì œê±°

            seen.add(char)

        return ''.join(stack)
